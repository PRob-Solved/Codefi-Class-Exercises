function greeting(message) {
  return function (name) {
    return message + " " + name;
  };
}

let sayHello = greeting("Hello");
console.log(sayHello("John"));

Exercise 3: Intermediate

Aim: Create a function with the drinks array as an argument and return the drinks objects sorted by price in ascending order. 

const drinks = [
  { name: "lemonade", price: 50 },
  { name: "lime", price: 10 },
];

function priceSort(drinksArray) {
  drinksArray.sort((a, b) => a.price - b.price);
}

priceSort(drinks);
console.log(drinks);

Exercise 4: Intermediate

Aim: Create the function that takes an array with objects and returns the sum of people's budgets. 

const myArray = [
  { name: "John", age: 21, budget: 23000 },
  { name: "Steve", age: 32, budget: 40000 },
  { name: "Martin", age: 16, budget: 2700 },
];

const sum = myArray.map((element) => element.budget).reduce((a, b) => a + b, 0);
console.log(sum);
